{ Algoritimo para solução do problema dos grãos de trigo no tabuleiro de
  xadrez.

  Da subrotina retira_dec(real) espera-se o no. real passado como
  parâmetro decrescido de sua parte decimal. Uma equivalente a floor e
  suas outras possibilidades da math.h do C.

  Algoritimo por Raimundo de Araujo Pereira Santos
  68345-7
  Física Computacional
}

VARIAVEIS:

massa_grao, massa_saco, qtde_graos_saco, massa_graos,
qtde_graos, qtde_sacos, qtde_sacosr, massa_sacos, i : reais

INICIO

 { inicializando variáveis }

 massa_grao <- 0.00032
 massa_saco <- 0.3
 massa_graos <- 0.0
 massa_sacos <- 0.0
 qtde_graos_saco <- 60.0 / massa_grao
 qtde_graos <- 1.0
 qtde_sacos <- 0.0
 qtde_sacosr <- 0.0 { arredondado }
 i <- 0.0 { contador }

 PARA i ATE 63 i <- i + 1 FACA

 qtde_graos <- qtde_graos * 2

 FIM_PARA

 qtde_graos <- qtde_graos - 1
 massa_graos = massa_grao * qtde_graos

 qtde_sacos <- qtde_graos / qtde_graos_saco

 SE (qtde_sacos - retira_dec(qtde_sacos)) != 0 ENTAO
  
  { arredonda para cima, pois retira_dec() o faz para baixo }
  qtde_sacosr = retira_dec(qtde_sacos) + 1
 
 SENAO qtde_sacosr <- qtde_sacos

 FIM_SE

 massa_sacos <- massa_graos + (qtde_sacos * massa_saco)

 IMPRIMA(qtde_graos, qtde_sacosr, massa_graos, massa_sacos)

FIM { programa }
